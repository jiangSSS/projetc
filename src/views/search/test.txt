CREATE TABLE `sys_org_user` (
  `user_id` bigint(20) NOT NULL AUTO_INCREMENT,
  `emp_id` bigint(10) DEFAULT NULL,
  `comp_id` bigint(10) DEFAULT NULL,
  `org_id` bigint(20) DEFAULT NULL,
  `login_id` varchar(50) DEFAULT NULL COMMENT '用户登录名',
  `password` varchar(50) DEFAULT NULL COMMENT '用户密码',
  `name` varchar(50) DEFAULT NULL COMMENT '姓名',
  `adm_flag` int(1) DEFAULT '0' COMMENT '1-管理员 0-普通用户',
  `adm_orgid` bigint(20) DEFAULT NULL,
  `dis_flag` int(1) NOT NULL DEFAULT '0' COMMENT '禁用标识：0-有效(默认) 1-禁用',
  `effect_date` date DEFAULT NULL COMMENT '生效日期',
  `expiry_date` date DEFAULT NULL COMMENT '失效日期',
  `del_flag` int(1) NOT NULL DEFAULT '0' COMMENT '删除标识：0-未删除(默认) 1-已删除',
  `create_uid` bigint(20) DEFAULT NULL,
  `create_tm` datetime DEFAULT NULL,
  `update_uid` bigint(20) DEFAULT NULL,
  `update_tm` datetime DEFAULT NULL,
  `head_img` varchar(200) DEFAULT NULL COMMENT '头像地址',
  `nick_name` varchar(50) DEFAULT NULL COMMENT '昵称',
  `comp_name` varchar(200) DEFAULT NULL COMMENT '公司名称',
  `job` varchar(50) DEFAULT NULL COMMENT '职位',
  `industry` varchar(50) DEFAULT NULL COMMENT '行业',
  PRIMARY KEY (`user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=103 DEFAULT CHARSET=utf8 COMMENT='用户表';
